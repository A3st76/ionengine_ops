7class D3DPipeline : public Pipeline {
8public:
9
10 D3DPipeline(winrt::com_ptr<ID3D12Device4>& device, const GraphicsPipelineDesc& pipeline_desc)
11 : m_device(device),
12 m_pipeline_type(PipelineType::Graphics),
13 m_d3d12_root_signature(static_cast<D3DDescriptorSetLayout&>(pipeline_desc.layout.get()).get_d3d12_root_signature()) {
14
15 // Input Element Description
16 // Description of the input data in the pipeline
17 std::vector<D3D12_INPUT_ELEMENT_DESC> input_descs;
18 for(auto& input : pipeline_desc.inputs) {
19 D3D12_INPUT_ELEMENT_DESC input_desc{};
20 input_desc.SemanticName = input.semantic_name.c_str();
21 input_desc.SemanticIndex = input.index;
22 input_desc.Format = static_cast<DXGI_FORMAT>(input.format);
23 input_desc.InputSlot = input.slot;
24 input_desc.AlignedByteOffset = D3D12_APPEND_ALIGNED_ELEMENT;
25 input_desc.InputSlotClass = D3D12_INPUT_CLASSIFICATION_PER_VERTEX_DATA;
26 input_desc.InstanceDataStepRate = 0;
27
28 input_descs.emplace_back(input_desc);
29 }
30
31 D3D12_INPUT_LAYOUT_DESC input_layout_desc{};
32 input_layout_desc.pInputElementDescs = input_descs.data();
33 input_layout_desc.NumElements = static_cast<uint32>(input_descs.size());
34
35 // Rasterizer Description
36 // Description of pipeline rasterization
37 D3D12_RASTERIZER_DESC rasterizer_desc{};
38 rasterizer_desc.FillMode = convert_fill_mode(pipeline_desc.rasterizer.fill_mode);
39 rasterizer_desc.CullMode = convert_cull_mode(pipeline_desc.rasterizer.cull_mode);
40 rasterizer_desc.FrontCounterClockwise = false;
41 rasterizer_desc.DepthBias = pipeline_desc.rasterizer.depth_bias;
42 rasterizer_desc.DepthBiasClamp = 0.0f;
43 rasterizer_desc.SlopeScaledDepthBias = 0.0f;
44 rasterizer_desc.DepthClipEnable = true;
45 rasterizer_desc.MultisampleEnable = false;
46 rasterizer_desc.AntialiasedLineEnable = false;
47 rasterizer_desc.ForcedSampleCount = 0;
48 rasterizer_desc.ConservativeRaster = D3D12_CONSERVATIVE_RASTERIZATION_MODE_OFF;
49
50 // Depth Stencil Description
51 // Description of the depth and stencil of the pipeline
52 D3D12_DEPTH_STENCILOP_DESC front_face_desc{};
53 front_face_desc.StencilFailOp = convert_stencil_op(pipeline_desc.depth_stencil.front_face.fail_op);
54 front_face_desc.StencilPassOp = convert_stencil_op(pipeline_desc.depth_stencil.front_face.pass_op);
55 front_face_desc.StencilDepthFailOp = convert_stencil_op(pipeline_desc.depth_stencil.front_face.depth_fail_op);
56 front_face_desc.StencilFunc = convert_comparison_func(pipeline_desc.depth_stencil.front_face.func);
57
58 D3D12_DEPTH_STENCILOP_DESC back_face_desc{};
59 back_face_desc.StencilFailOp = convert_stencil_op(pipeline_desc.depth_stencil.back_face.fail_op);
60 back_face_desc.StencilPassOp = convert_stencil_op(pipeline_desc.depth_stencil.back_face.pass_op);
61 back_face_desc.StencilDepthFailOp = convert_stencil_op(pipeline_desc.depth_stencil.back_face.depth_fail_op);
62 back_face_desc.StencilFunc = convert_comparison_func(pipeline_desc.depth_stencil.back_face.func);
63
64 D3D12_DEPTH_STENCIL_DESC depth_stencil_desc{};
65 depth_stencil_desc.DepthEnable = pipeline_desc.depth_stencil.depth_test_enable;
66 depth_stencil_desc.DepthWriteMask = pipeline_desc.depth_stencil.depth_write_enable ? D3D12_DEPTH_WRITE_MASK_ALL : D3D12_DEPTH_WRITE_MASK_ZERO;
67 // depth_stencil_desc.DepthBoundsTestEnable = desc.depth_stencil.depth_bounds_test_enable;
68 depth_stencil_desc.DepthFunc = convert_comparison_func(pipeline_desc.depth_stencil.depth_func);
69 depth_stencil_desc.StencilEnable = pipeline_desc.depth_stencil.stencil_enable;
70 depth_stencil_desc.StencilReadMask = pipeline_desc.depth_stencil.stencil_read_mask;
71 depth_stencil_desc.StencilWriteMask = pipeline_desc.depth_stencil.stencil_write_mask;
72 depth_stencil_desc.FrontFace = front_face_desc;
73 depth_stencil_desc.BackFace = back_face_desc;
74
75 // Get Render Pass Description
76 auto& render_pass_desc = pipeline_desc.render_pass.get().get_desc();
77
78 // Blend Description
79 // Description of pipeline blend
80 D3D12_BLEND_DESC blend_desc{};
81 for(uint32 i = 0; i < render_pass_desc.colors.size(); ++i) {
82 if(render_pass_desc.colors[i].format == Format::Unknown) {
83 continue;
84 }
85 auto& render_target = blend_desc.RenderTarget[i];
86
87 render_target.BlendEnable = pipeline_desc.blend.blend_enable;
88 render_target.BlendOp = convert_blend_op(pipeline_desc.blend.blend_op);
89 render_target.SrcBlend = convert_blend(pipeline_desc.blend.blend_src);
90 render_target.DestBlend = convert_blend(pipeline_desc.blend.blend_dest);
91 render_target.BlendOpAlpha = convert_blend_op(pipeline_desc.blend.blend_op_alpha);
92 render_target.SrcBlendAlpha = convert_blend(pipeline_desc.blend.blend_src_alpha);
93 render_target.DestBlendAlpha = convert_blend(pipeline_desc.blend.blend_dest_alpha);
94 render_target.RenderTargetWriteMask = D3D12_COLOR_WRITE_ENABLE_ALL;
95 }
96
97 // Render Target Formats
98 std::vector<DXGI_FORMAT> rtv_formats;
99 rtv_formats.resize(render_pass_desc.colors.size());
100 for(uint32 i = 0; i < render_pass_desc.colors.size(); ++i) {
101 if(render_pass_desc.colors[i].format == Format::Unknown) {
102 continue;
103 }
104 rtv_formats[i] = static_cast<DXGI_FORMAT>(render_pass_desc.colors[i].format);
105 }
106
107 // Sample Description
108 DXGI_SAMPLE_DESC sample_desc{};
109 sample_desc.Count = render_pass_desc.sample_count;
110
111 // Graphics Pipeline State Description
112 // Description of the pipeline
113 D3D12_GRAPHICS_PIPELINE_STATE_DESC graphics_pipeline_desc{};
114 graphics_pipeline_desc.pRootSignature = m_d3d12_root_signature.get().get();
115 graphics_pipeline_desc.InputLayout = input_layout_desc;
116 graphics_pipeline_desc.PrimitiveTopologyType = D3D12_PRIMITIVE_TOPOLOGY_TYPE_TRIANGLE;
117 graphics_pipeline_desc.RasterizerState = rasterizer_desc;
118 graphics_pipeline_desc.DepthStencilState = depth_stencil_desc;
119 graphics_pipeline_desc.BlendState = blend_desc;
120 for(auto& shader : pipeline_desc.shaders) {
121 switch(shader.get().get_type()) {
122 case ShaderType::Vertex: graphics_pipeline_desc.VS = static_cast<D3DShader&>(shader.get()).get_d3d12_shader(); break;
123 case ShaderType::Pixel: graphics_pipeline_desc.PS = static_cast<D3DShader&>(shader.get()).get_d3d12_shader(); break;
124 case ShaderType::Geometry: graphics_pipeline_desc.GS = static_cast<D3DShader&>(shader.get()).get_d3d12_shader(); break;
125 case ShaderType::Domain: graphics_pipeline_desc.DS = static_cast<D3DShader&>(shader.get()).get_d3d12_shader(); break;
126 case ShaderType::Hull: graphics_pipeline_desc.HS = static_cast<D3DShader&>(shader.get()).get_d3d12_shader(); break;
127 }
128 }
129 std::memcpy(graphics_pipeline_desc.RTVFormats, rtv_formats.data(), rtv_formats.size() * sizeof(DXGI_FORMAT));
130 graphics_pipeline_desc.NumRenderTargets = static_cast<uint32>(rtv_formats.size());
131 graphics_pipeline_desc.DSVFormat = static_cast<DXGI_FORMAT>(render_pass_desc.depth_stencil.format);
132 graphics_pipeline_desc.SampleDesc = sample_desc;
133
134 ASSERT_SUCCEEDED(m_device.get()->CreateGraphicsPipelineState(&graphics_pipeline_desc, __uuidof(ID3D12PipelineState), m_d3d12_pipeline_state.put_void()));
135 }
136
137 PipelineType get_type() const override { return m_pipeline_type; }
138
139 winrt::com_ptr<ID3D12RootSignature>& get_d3d12_root_signature() { return m_d3d12_root_signature; }
140 winrt::com_ptr<ID3D12PipelineState>& get_d3d12_pipeline_state() { return m_d3d12_pipeline_state; }
141
142private:
143
144 std::reference_wrapper<winrt::com_ptr<ID3D12Device4>> m_device;
145 std::reference_wrapper<winrt::com_ptr<ID3D12RootSignature>> m_d3d12_root_signature;
146
147 winrt::com_ptr<ID3D12PipelineState> m_d3d12_pipeline_state;
148
149 PipelineType m_pipeline_type;
150};
