67 void resource_barriers(const std::vector<ResourceBarrierDesc>& barrier_descs) override {
68
69 std::vector<D3D12_RESOURCE_BARRIER> d3d12_barriers;
70 for(auto& barrier_desc : barrier_descs) {
71
72 D3D12_RESOURCE_BARRIER resource_barrier{};
73 resource_barrier.Type = D3D12_RESOURCE_BARRIER_TYPE_TRANSITION;
74 resource_barrier.Transition.pResource = static_cast<D3DResource&>(barrier_desc.resource.get()).get_d3d12_resource().get();
75 resource_barrier.Transition.Subresource = D3D12_RESOURCE_BARRIER_ALL_SUBRESOURCES;
76 resource_barrier.Transition.StateAfter = convert_resource_state(barrier_desc.state_after);
77 resource_barrier.Transition.StateBefore = convert_resource_state(barrier_desc.state_before);
78
79 d3d12_barriers.emplace_back(resource_barrier);
80 }
81
82 if(!d3d12_barriers.empty()) {
83 m_d3d12_command_list->ResourceBarrier(static_cast<uint32>(d3d12_barriers.size()), d3d12_barriers.data());
84 }

87 void begin_render_pass(RenderPass& render_pass, FrameBuffer& frame_buffer, const ClearValueDesc& clear_value_desc) override {
88
89 auto& render_pass_desc = static_cast<D3DRenderPass&>(render_pass).get_desc();
90 auto& colors = static_cast<D3DFrameBuffer&>(frame_buffer).get_desc().colors;
91 auto& depth_stencil = static_cast<D3DFrameBuffer&>(frame_buffer).get_desc().depth_stencil;
92
93 std::vector<D3D12_RENDER_PASS_RENDER_TARGET_DESC> render_pass_rt_descs;
94 for(uint32 i = 0; i < colors.size(); ++i) {
95
96 D3D12_RENDER_PASS_BEGINNING_ACCESS begin{};
97 begin.Type = convert_render_pass_type(render_pass_desc.colors[i].load_op);
98
99 begin.Clear.ClearValue.Format = static_cast<DXGI_FORMAT>(render_pass_desc.colors[i].format);
100
101 begin.Clear.ClearValue.Color[0] = clear_value_desc.colors[i].r;
102 begin.Clear.ClearValue.Color[1] = clear_value_desc.colors[i].g;
103 begin.Clear.ClearValue.Color[2] = clear_value_desc.colors[i].b;
104 begin.Clear.ClearValue.Color[3] = clear_value_desc.colors[i].a;
105
106 D3D12_RENDER_PASS_ENDING_ACCESS end{};
107 end.Type = convert_render_pass_type(render_pass_desc.colors[i].store_op);
108
109 D3D12_RENDER_PASS_RENDER_TARGET_DESC render_pass_rt_desc;
110 render_pass_rt_desc.cpuDescriptor = static_cast<D3DView&>(colors[i].get()).get_d3d12_cpu_descriptor();
111 render_pass_rt_desc.BeginningAccess = begin;
112 render_pass_rt_desc.EndingAccess = end;
113
114 render_pass_rt_descs.emplace_back(render_pass_rt_desc);
115 }
116
117 D3D12_RENDER_PASS_DEPTH_STENCIL_DESC render_pass_ds_desc;
118 if(depth_stencil.has_value()) {
119
120 D3D12_RENDER_PASS_BEGINNING_ACCESS depth_begin{};
121 depth_begin.Type = convert_render_pass_type(render_pass_desc.depth_stencil.depth_load_op);
122 depth_begin.Clear.ClearValue.Format = static_cast<DXGI_FORMAT>(render_pass_desc.depth_stencil.format);
123 depth_begin.Clear.ClearValue.DepthStencil.Depth = clear_value_desc.depth;
124
125 D3D12_RENDER_PASS_ENDING_ACCESS depth_end{};
126 depth_end.Type = convert_render_pass_type(render_pass_desc.depth_stencil.depth_store_op);
127
128 D3D12_RENDER_PASS_BEGINNING_ACCESS stencil_begin{};
129 stencil_begin.Type = convert_render_pass_type(render_pass_desc.depth_stencil.stencil_load_op);
130 stencil_begin.Clear.ClearValue.Format = static_cast<DXGI_FORMAT>(render_pass_desc.depth_stencil.format);
131 stencil_begin.Clear.ClearValue.DepthStencil.Stencil = clear_value_desc.stencil;
132
133 D3D12_RENDER_PASS_ENDING_ACCESS stencil_end{};
134 stencil_end.Type = convert_render_pass_type(render_pass_desc.depth_stencil.stencil_store_op);
135
136 render_pass_ds_desc.cpuDescriptor = static_cast<D3DView&>(depth_stencil.value().get()).get_d3d12_cpu_descriptor();
137 render_pass_ds_desc.DepthBeginningAccess = depth_begin;
138 render_pass_ds_desc.DepthEndingAccess = depth_end;
139 render_pass_ds_desc.StencilBeginningAccess = stencil_begin;
140 render_pass_ds_desc.StencilEndingAccess = stencil_end;
141
142 m_d3d12_command_list->BeginRenderPass(static_cast<uint32>(render_pass_rt_descs.size()), render_pass_rt_descs.data(), &render_pass_ds_desc, D3D12_RENDER_PASS_FLAG_NONE);
143 } else {
144 m_d3d12_command_list->BeginRenderPass(static_cast<uint32>(render_pass_rt_descs.size()), render_pass_rt_descs.data(), nullptr, D3D12_RENDER_PASS_FLAG_NONE);
145 }
