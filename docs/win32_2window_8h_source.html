16

20
21 m_wnd = CreateWindow(
22 wnd_class.lpszClassName,
23 stws(label).c_str(),
24 static_cast<uint32>(window_style),
25 0, 100,
26 width, height,
27 nullptr, nullptr,
28 wnd_class.hInstance,

105 static LRESULT window_procedure(HWND hWnd, UINT msg, WPARAM wParam, LPARAM lParam) {
106
107 WindowsWindow* window = reinterpret_cast<WindowsWindow*>(GetWindowLongPtr(hWnd, GWLP_USERDATA));
108 WindowEventHandler event_handler{};
109
110 if (window) {
111 event_handler.window_id = window->m_id;
112 }
113
114 switch(msg) {
115
116 case WM_CLOSE: {
117 event_handler.event_type = WindowEvent::Closed;
118 window->m_event_loop.get().emplace_event(event_handler);
119 break;
120 }
121 case WM_SIZE: {
122 event_handler.event_type = WindowEvent::Sized;
123 event_handler.event = PhysicalSize { LOWORD(lParam), HIWORD(lParam) };
124 window->m_event_loop.get().emplace_event(event_handler);
125 break;
126 }
127 case WM_MOUSEMOVE: {
128 event_handler.event_type = WindowEvent::Moved;
129 event_handler.event = MouseMoved { GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam), false };
130 window->m_event_loop.get().emplace_event(event_handler);
131 break;
132 }
133 case WM_INPUT: {
134 byte buffer[48];
135 uint32 size = 0;
136
137 GetRawInputData(reinterpret_cast<HRAWINPUT>(lParam), RID_INPUT, nullptr, &size, sizeof(RAWINPUTHEADER));
138 if (size <= 48) {
139 GetRawInputData(reinterpret_cast<HRAWINPUT>(lParam), RID_INPUT, buffer, &size, sizeof(RAWINPUTHEADER));
140 }
141
142 RAWINPUT* raw_input = reinterpret_cast<RAWINPUT*>(buffer);
143 DWORD device_type = raw_input->header.dwType;
144
145 if (device_type == RIM_TYPEKEYBOARD) {
146 event_handler.event_type = WindowEvent::KeyboardInput;
147
148 USHORT key = raw_input->data.keyboard.VKey;
149 USHORT flags = raw_input->data.keyboard.Flags;
150 UINT msg = raw_input->data.keyboard.Message;
151
152 switch (msg) {
153 case WM_KEYDOWN:
154 case WM_SYSKEYDOWN: event_handler.event = KeyboardInput { static_cast<uint32>(key), ElementState::Pressed }; break;
155 case WM_KEYUP:
156 case WM_SYSKEYUP: event_handler.event = KeyboardInput { static_cast<uint32>(key), ElementState::Released }; break;
157 }
158
159 window->m_event_loop.get().emplace_event(event_handler);
160 } else if (device_type == RIM_TYPEMOUSE) {
161 event_handler.event_type = WindowEvent::MouseInput;
162
163 LONG x = raw_input->data.mouse.lLastX;
164 LONG y = raw_input->data.mouse.lLastY;
165 ULONG buttons = raw_input->data.mouse.ulButtons;
166 USHORT flags = raw_input->data.mouse.usFlags;
167 USHORT button_flags = raw_input->data.mouse.usButtonFlags;
168 USHORT button_data = raw_input->data.mouse.usButtonData;
169
170 switch (buttons) {
171 case RI_MOUSE_LEFT_BUTTON_DOWN: event_handler.event = MouseInput { MouseButton::Left, ElementState::Pressed }; break;
172 case RI_MOUSE_RIGHT_BUTTON_DOWN: event_handler.event = MouseInput { MouseButton::Right, ElementState::Pressed }; break;
173 case RI_MOUSE_MIDDLE_BUTTON_DOWN: event_handler.event = MouseInput { MouseButton::Middle, ElementState::Pressed }; break;
174 case RI_MOUSE_BUTTON_4_DOWN: event_handler.event = MouseInput { MouseButton::Four, ElementState::Pressed }; break;
175 case RI_MOUSE_BUTTON_5_DOWN: event_handler.event = MouseInput { MouseButton::Five, ElementState::Pressed }; break;
176 case RI_MOUSE_LEFT_BUTTON_UP: event_handler.event = MouseInput { MouseButton::Left, ElementState::Released }; break;
177 case RI_MOUSE_RIGHT_BUTTON_UP: event_handler.event = MouseInput { MouseButton::Right, ElementState::Released }; break;
178 case RI_MOUSE_MIDDLE_BUTTON_UP: event_handler.event = MouseInput { MouseButton::Middle, ElementState::Released }; break;
179 case RI_MOUSE_BUTTON_4_UP: event_handler.event = MouseInput { MouseButton::Four, ElementState::Released }; break;
180 case RI_MOUSE_BUTTON_5_UP: event_handler.event = MouseInput { MouseButton::Five, ElementState::Released }; break;
181 }
182
183 window->m_event_loop.get().emplace_event(event_handler);
184
185 if (button_flags & RI_MOUSE_WHEEL) {
186 event_handler.event_type = WindowEvent::MouseWheel;
187 event_handler.event = MouseWheel { static_cast<float>(button_data / WHEEL_DELTA) };
188 window->m_event_loop.get().emplace_event(event_handler);
189 }
190
191 event_handler.event_type = WindowEvent::MouseMoved;
192 event_handler.event = MouseMoved { x, y, true };
193 window->m_event_loop.get().emplace_event(event_handler);
194 }
195 break;
196 }
197 }
198 return DefWindowProc(hWnd, msg, wParam, lParam);
199 }
