105

108 ResourceState resource_state = ResourceState::Common;

113 ASSERT_SUCCEEDED(m_device.get()->CreatePlacedResource(

116 &std::get<D3D12_RESOURCE_DESC>(m_d3d12_resource_desc),
117 convert_resource_state(resource_state),
118 nullptr,
119 __uuidof(ID3D12Resource), m_d3d12_resource.put_void()
120 ));
121 }
