98 }

100 auto& resource_desc = std::get<D3D12_RESOURCE_DESC>(m_resource.get().get_d3d12_desc());
101
102 D3D12_CONSTANT_BUFFER_VIEW_DESC cbv_view{};

110 auto& sampler_desc = std::get<D3D12_SAMPLER_DESC>(m_resource.get().get_d3d12_desc());

112 break;
